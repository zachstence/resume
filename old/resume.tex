% Separator
\newcommand{\sep}{\ \textbf{|} \ }

% Heading for a new resume section
\newcommand{\resheading}[1]{
    \begin{mdframed}[
        backgroundcolor=black!15,
        linewidth=\headingborderwidth,
        innertopmargin=4pt,
        innerbottommargin=4pt,
        skipabove=6pt,
        skipbelow=2pt]
        \textbf{\large #1}
    \end{mdframed}
}

% Sub heading for a resume section with 2 bolded lines and an italic line on the left, and two normal lines on the right
% Intended to be used with a double major education item, the arguments are major 1, date range, major 2, location, university name
\newcommand{\dressubheading}[5]{
    \textbf{#1} \hfill #2 \\[0pt]
    \textbf{#3} \hfill #4 \\[0pt]
    \textit{#5} \\[0pt]
}

% Sub heading for a resume section with a bolded and italic line on the left, and two normal lines on the right
% Intended to be used with an experience item, the arguments are position, date range, company name, location
\newcommand{\ressubheading}[4]{
    \textbf{#1} \hfill #2 \\[0pt]
    \textit{#3} \hfill #4 \\[0pt]
}

% Bolded item for a resume section to be used in an itemize. Bolds the item text (argument 1) and adds a right-aligned date range (argument 2)
\newcommand{\resdateitem}[2]{
	\item \textbf{#1} \hfill #2 \\
}

% Adapted itemize environment with decreased spacing
\newenvironment{resitemize}{
    \vspace{-6pt}
    \begin{itemize}
    \setlength\itemsep{-2pt}
}{
    \end{itemize}
}
